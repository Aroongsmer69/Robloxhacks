-- Client-side script to change the skybox
local Lighting = game:GetService("Lighting")

-- Clear existing skies first (to prevent duplicates)
for _, existingSky in ipairs(Lighting:GetChildren()) do
    if existingSky:IsA("Sky") then
        existingSky:Destroy()
    end
end

-- Create new skybox with your custom decal
local newSky = Instance.new("Sky")
local decalId = "rbxassetid://132830820075183"

-- Apply the decal to all 6 faces of the skybox
newSky.SkyboxBk = decalId  -- Back
newSky.SkyboxDn = decalId  -- Down
newSky.SkyboxFt = decalId  -- Front
newSky.SkyboxLf = decalId  -- Left
newSky.SkyboxRt = decalId  -- Right
newSky.SkyboxUp = decalId  -- Up

-- Optional: Adjust sky properties for better appearance
newSky.StarCount = 0       -- Remove stars since we're using a custom sky
newSky.SunAngularSize = 0  -- Remove sun
newSky.MoonAngularSize = 0 -- Remove moon

newSky.Parent = Lighting
