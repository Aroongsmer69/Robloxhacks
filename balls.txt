-- Client-side script (LocalScript) that plays a sound very loud on loop
local soundId = "rbxassetid://70700889250445"

-- Create sound object
local sound = Instance.new("Sound")
sound.SoundId = soundId
sound.Volume = 10  -- Max volume (even beyond normal max of 10)
sound.Looped = true
sound.PlayOnRemove = true
sound.Parent = game:GetService("SoundService")

-- Play the sound
sound:Play()

-- Keep the sound playing even if the script is destroyed
sound.Parent = game:GetService("SoundService")

-- Optional: Make it even louder by adjusting the SoundService properties
game:GetService("SoundService").Volume = 1.5  -- This will amplify all sounds

-- Optional: Print confirmation
print("Playing loud sound on loop:", soundId)
